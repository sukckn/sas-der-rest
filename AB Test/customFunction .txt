method getABSplitGroups(double numRows, double numControl, double numChall_1, double numChall_2, double numChall_3, double numChall_4, double numChall_5) returns varchar;
  dcl package pymas py;
  dcl double pystop;
  dcl package logger logr('App.tk.MAS');

  dcl nvarchar(10485760) pypgm testRows;
  dcl double rc;
  dcl double revision;
  if null(py) and pystop ^= 1 then do;
    py= _new_ pymas();
    if inmas() then do;
      rc= py.useModule('"getABSplitGroups-397911658"', 1);
    end;
    else do;
      rc= 1;
    end;
    /********************* Python code *********************************/
    if rc then do;
      rc = py.appendSrcLine('import secrets');
      rc = py.appendSrcLine('import collections');
      rc = py.appendSrcLine('');
      rc = py.appendSrcLine('def getABSplitGroupsPy (numRows,numControl,numChall_1,numChall_2,numChall_3,numChall_4,numChall_5):');
      rc = py.appendSrcLine('    ''Output:testRows''');
      rc = py.appendSrcLine('    ');
      rc = py.appendSrcLine('    challList= [numChall_1, numChall_2, numChall_3, numChall_4, numChall_5]');
      rc = py.appendSrcLine('    challName= [''Chall_1'', ''Chall_2'', ''Chall_3'', ''Chall_4'', ''Chall_5'']');
      rc = py.appendSrcLine('    testRowsGroups= {}');
      rc = py.appendSrcLine('');
      rc = py.appendSrcLine('    allRows= list(range(numRows))');
      rc = py.appendSrcLine('    ');
      rc = py.appendSrcLine('    #set the rows for the control group');
      rc = py.appendSrcLine('    for i in range(numControl):');
      rc = py.appendSrcLine('        row= secrets.choice(allRows)');
      rc = py.appendSrcLine('        allRows.remove(row)');
      rc = py.appendSrcLine('        testRowsGroups[row]= ''Ctrl''');
      rc = py.appendSrcLine('    ');
      rc = py.appendSrcLine('    #set the rows for the challenger group');
      rc = py.appendSrcLine('    challNameIdx= 0');
      rc = py.appendSrcLine('    for chall in challList:');
      rc = py.appendSrcLine('        for c in range(chall):');
      rc = py.appendSrcLine('            row= secrets.choice(allRows)');
      rc = py.appendSrcLine('            allRows.remove(row)');
      rc = py.appendSrcLine('            testRowsGroups[row]= challName[challNameIdx]');
      rc = py.appendSrcLine('        challNameIdx+= 1');
      rc = py.appendSrcLine('   ');
      rc = py.appendSrcLine('    #set the rows for the champion group');
      rc = py.appendSrcLine('    for i in allRows:');
      rc = py.appendSrcLine('        testRowsGroups[i]= ''Champ''');
      rc = py.appendSrcLine('        ');
      rc = py.appendSrcLine('    #order result by row number');
      rc = py.appendSrcLine('    testRows_Ordered= collections.OrderedDict(sorted(testRowsGroups.items()))');
      rc = py.appendSrcLine('    ');
      rc = py.appendSrcLine('    #we want to return the result in a comma separated string');
      rc = py.appendSrcLine('    testRows= ''''');
      rc = py.appendSrcLine('    for row in testRows_Ordered:');
      rc = py.appendSrcLine('        testRows+= testRows_Ordered[row] + '',''');
      rc = py.appendSrcLine('        ');
      rc = py.appendSrcLine('    testRows= testRows[0:len(testRows)-1]');
      rc = py.appendSrcLine('    return testRows');
      pypgm = py.getSource();
      
      revision = py.publish(pypgm, '"getABSplitGroups-397911658"');
      if revision < 1 then do;
        pystop = 1;
        logr.log( 'e', 'publish revision=$s', revision );
        return '';
      end;
    end;
    else do;
      logr.log( 'd', 'useModule rc=$s', rc );
    end;
    rc = py.useMethod('getABSplitGroupsPy');
    if rc then do;
      pystop = 1;
      logr.log( 'e', 'useMethod rc=$s', rc );
      return '';
    end;
  end;
  if pystop ^= 1 then do;
    rc= py.setInt ('numRows', numRows);
    if rc then do;
      logr.log( 'e', 'set numRows rc=$s', rc );
      return '';
    end;
    rc= py.setInt ('numControl', numControl);
    if rc then do;
      logr.log( 'e', 'set numControl rc=$s', rc );
      return '';
    end;
    rc= py.setInt ('numChall_1', numChall_1);
    if rc then do;
      logr.log( 'e', 'set numChall_1 rc=$s', rc );
      return '';
    end;
    rc= py.setInt ('numChall_2', numChall_2);
    if rc then do;
      logr.log( 'e', 'set numChall_2 rc=$s', rc );
      return '';
    end;
    rc= py.setInt ('numChall_3', numChall_3);
    if rc then do;
      logr.log( 'e', 'set numChall_3 rc=$s', rc );
      return '';
    end;
    rc= py.setInt ('numChall_4', numChall_4);
    if rc then do;
      logr.log( 'e', 'set numChall_4 rc=$s', rc );
      return '';
    end;
    rc= py.setInt ('numChall_5', numChall_5);
    if rc then do;
      logr.log( 'e', 'set numChall_5 rc=$s', rc );
      return '';
    end;
    rc= py.execute();
    logr.log( 'd', 'getABSplitGroupsPy rc=$s', rc );
    testRows = py.getString('testRows');
  end;
  
  return testRows;
end;
